  function startTraining() {
    show('screen-module');

  function show(id) { document.getElementById(id).classList.add('active'); }
  function hide(id) { document.getElementById(id).classList.remove('active'); }

  function renderModule(idx) {
    currentModule = idx;
    quizAnswered = false;
    const m = modules[idx];
    const total = modules.length;
    const pct = Math.round((idx / total) * 100);

    document.getElementById('top-tag').innerHTML = (idx+1) + ' de ' + total + '<br>módulos';
    document.getElementById('prog-label').textContent = m.tag;
    document.getElementById('prog-pct').textContent = pct + '%';
    document.getElementById('prog-fill').style.width = pct + '%';

    const dots = document.getElementById('nav-dots');
    dots.innerHTML = modules.map((_,i) => {
      let c = 'dot';
      if (i < idx) c += ' done';
      if (i === idx) c += ' current';
      return '<div class="' + c + '"></div>';
    }).join('');

    const isLast = idx === total - 1;
    const btnLabel = isLast ? 'Finalizar e registrar 📝' : 'Próximo módulo →';

    document.getElementById('module-content').innerHTML = `
      <div class="card">
        <div class="module-tag">${m.tag}</div>
        <span class="module-emoji">${m.emoji}</span>
        <div class="module-title">${m.title}</div>
        <div class="module-sub">${m.sub}</div>
        <div class="divider"></div>
        ${m.content}
      </div>
      <div class="card">
        <div class="module-tag green">Quiz rápido</div>
        <div class="quiz-q">${m.quiz.q}</div>
        <div class="quiz-options" id="quiz-opts">
          ${m.quiz.opts.map((o,i) => `<button class="quiz-opt" onclick="answer(${i})">${o}</button>`).join('')}
        </div>
        <div class="quiz-feedback" id="quiz-fb"></div>
      </div>
      <button class="btn-primary" id="btn-next" disabled onclick="nextModule()">${btnLabel}</button>
    `;
    window.scrollTo({top:0, behavior:'smooth'});
  }

  function answer(idx) {
    if (quizAnswered) return;
    quizAnswered = true;
    const m = modules[currentModule];
    const opts = document.querySelectorAll('.quiz-opt');
    const fb = document.getElementById('quiz-fb');
    const ok = idx === m.quiz.correct;
    opts.forEach((o,i) => {
      o.disabled = true;
      if (i === m.quiz.correct) o.classList.add('correct');
      else if (i === idx) o.classList.add('wrong');
    });
    fb.textContent = (ok ? '✓ ' : '✗ ') + (ok ? m.quiz.ok : m.quiz.no);
    fb.className = 'quiz-feedback show ' + (ok ? 'ok' : 'no');
    scores[currentModule] = ok;
    document.getElementById('btn-next').disabled = false;
  }

  function nextModule() {
    if (currentModule < modules.length - 1) {
      renderModule(currentModule + 1);
    } else {
      hide('screen-module');
      show('screen-form');
      document.getElementById('top-tag').textContent = 'Registro de conclusão';
      window.scrollTo({top:0, behavior:'smooth'});
    }
  }

  function selectScore(btn, val) {
    document.querySelectorAll('.score-btn').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    scoreVal = val;
  }

  function toggleAgree() {
    agreed = !agreed;
    document.getElementById('agree-box').classList.toggle('checked', agreed);
    document.getElementById('agree-check').classList.toggle('checked', agreed);
  }

  function maskCPF(v) {
    v = v.replace(/\D/g,'').slice(0,11);
    if (v.length > 9) v = v.slice(0,3)+'.'+v.slice(3,6)+'.'+v.slice(6,9)+'-'+v.slice(9);
    else if (v.length > 6) v = v.slice(0,3)+'.'+v.slice(3,6)+'.'+v.slice(6);
    else if (v.length > 3) v = v.slice(0,3)+'.'+v.slice(3);
    return v;
  }

  function showErr(id, show) {
    document.getElementById(id).classList.toggle('show', show);
    const fid = 'f-' + id.slice(2);
    const fi = document.getElementById(fid);
    if (fi) fi.classList.toggle('error', show);
  }

  function gerarComprovante() {
    const nome = document.getElementById('f-nome').value.trim();
    const cpf = document.getElementById('f-cpf').value.trim();
    const admissao = document.getElementById('f-admissao').value;
    const cargo = document.getElementById('f-cargo').value;
    const unidade = document.getElementById('f-unidade').value;

    let ok = true;
    showErr('e-nome', !nome); if (!nome) ok = false;
    const cpfNum = cpf.replace(/\D/g,'');
    showErr('e-cpf', cpfNum.length !== 11); if (cpfNum.length !== 11) ok = false;
    showErr('e-admissao', !admissao); if (!admissao) ok = false;
    showErr('e-cargo', !cargo); if (!cargo) ok = false;
    showErr('e-unidade', !unidade); if (!unidade) ok = false;
    showErr('e-score', scoreVal === null); if (scoreVal === null) ok = false;
    showErr('e-agree', !agreed); if (!agreed) ok = false;
    if (!ok) return;

    const now = new Date();
    const dataStr = now.toLocaleDateString('pt-BR', {day:'2-digit', month:'long', year:'numeric'});
    const horaStr = now.toLocaleTimeString('pt-BR', {hour:'2-digit', minute:'2-digit'});
    const admStr = new Date(admissao+'T12:00:00').toLocaleDateString('pt-BR', {day:'2-digit', month:'long', year:'numeric'});
    const pct = Math.round((scoreVal/9)*100);
    const uid = 'BLR-' + now.getFullYear() + String(now.getMonth()+1).padStart(2,'0') + String(now.getDate()).padStart(2,'0') + '-' + Math.random().toString(36).slice(2,7).toUpperCase();

    let label;
    if (pct === 100) label = 'Aprovado com excelência — acertou tudo!';
    else if (pct >= 78) label = 'Aprovado — Muito bom';
    else label = 'Aprovado';

    document.getElementById('comp-id').textContent = uid;
    document.getElementById('comp-score-num').textContent = scoreVal+'/9';
    document.getElementById('comp-score-label').innerHTML = '<strong>'+label+'</strong>'+pct+'% de aproveitamento';
    document.getElementById('cv-nome').textContent = nome;
    document.getElementById('cv-cpf').textContent = cpf;
    document.getElementById('cv-cargo').textContent = cargo;
    document.getElementById('cv-unidade').textContent = 'Belari — ' + unidade;
    document.getElementById('cv-admissao').textContent = admStr;
    document.getElementById('cv-data').textContent = dataStr + ' às ' + horaStr;
    document.getElementById('sig-code').textContent = uid;
    document.getElementById('sig-nome').textContent = nome;

    const saving = document.getElementById('saving-overlay');
    saving.classList.add('show');
    document.querySelector('#screen-form .btn-primary').disabled = true;

    const payload = { nome, cpf, admissao: admStr, cargo, unidade, score: scoreVal+'/9', aproveitamento: pct+'%', dataHora: dataStr+' às '+horaStr, codigo: uid };

    fetch(APPS_SCRIPT_URL, {
      method: 'POST',
      mode: 'no-cors',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    }).finally(() => {
      saving.classList.remove('show');
      hide('screen-form');
      show('screen-comp');
      document.getElementById('top-tag').textContent = 'Comprovante emitido';
      window.scrollTo({top:0, behavior:'smooth'});
    });
  }

  function novoRegistro() {
    document.getElementById('f-nome').value = '';
    document.getElementById('f-cpf').value = '';
    document.getElementById('f-admissao').value = '';
    document.getElementById('f-cargo').value = '';
    document.getElementById('f-unidade').value = '';
    scoreVal = null; agreed = false;
    document.querySelectorAll('.score-btn').forEach(b => b.classList.remove('selected'));
    document.getElementById('agree-box').classList.remove('checked');
    document.getElementById('agree-check').classList.remove('checked');
    ['e-nome','e-cpf','e-admissao','e-cargo','e-unidade','e-score','e-agree'].forEach(id => showErr(id, false));
    document.querySelector('#screen-form .btn-primary').disabled = false;
    hide('screen-comp');
    show('screen-form');
    window.scrollTo({top:0, behavior:'smooth'});
  }
